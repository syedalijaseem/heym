# Runbook — Self-Driving Animated Demo Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add a "Runbook" — a self-driving, animated demo that builds and runs a real, credential-free workflow (`textInput → wait → consoleLog`) from three entry points (Ctrl+K, the Guidelines/showcase panel, and an empty-canvas starter).

**Architecture:** A single `useRunbookPlayer` composable orchestrates everything. Menu entries create a real workflow named "Runbook — Heym demo" and navigate to its editor with a `?runbook=play` query flag; the editor detects that flag on mount and runs the choreography. The empty-canvas entry runs the same choreography in place. The choreography sequentially `addNode`s the three nodes with a slide-in-from-left entrance animation, connects them, sets the input value, and calls the real `executeWorkflow()`. The canvas auto-fits on every node-count change, so no manual viewport handling is needed.

**Tech Stack:** Vue 3 (`<script setup>` + Composition API), Pinia (`useWorkflowStore`), Vue Flow, TypeScript (strict), Tailwind; backend FastAPI + `WorkflowExecutor` for the contract test (pytest / unittest).

## Decision notes (locked)

- Demo workflow is **kept** (clearly named "Runbook — Heym demo"); a fresh one is created per menu trigger.
- consoleLog message uses the **expression** `$input.text.upper()` (DSL supports `.upper()`).
- The 3 nodes' full data is defined **inline** in `runbookScript.ts` (not via the canvas's `getDefaultNodeData`), to keep the demo self-contained and avoid a broad refactor of that 90-line factory.
- Navigation timing for menu entries is handled with a `?runbook=play` query flag consumed by `EditorView` on mount (no cross-route polling).

        {"id": "e2", "source": "wait1", "target": "log1"},

        joined = "\n".join(captured.output)
        self.assertIn(RUNBOOK_EXPECTED_LOG, joined)

    unittest.main()

```ts
import type { NodeType, WorkflowNode, WorkflowEdge } from "@/types/workflow";

/** Sample input the demo types into the Input node. */
export const RUNBOOK_INPUT_TEXT = "Heym is an ai native automation platform";

/** consoleLog message expression — logs the input upper-cased. DSL supports .upper(). */
export const RUNBOOK_LOG_EXPRESSION = "$input.text.upper()";

/** The demo workflow name created from menu entry points. */
export const RUNBOOK_WORKFLOW_NAME = "Runbook — Heym demo";

/** Query flag the editor consumes on mount to auto-play the runbook. */
export const RUNBOOK_QUERY_FLAG = "runbook";
export const RUNBOOK_QUERY_VALUE = "play";

/** Horizontal spacing between the three demo nodes. */
const NODE_DX = 320;

interface RunbookStepDef {
  type: NodeType;
  data: WorkflowNode["data"];
  /** Pause (ms) before this node slides in. */
  enterDelayMs: number;
}

/** The three real, credential-free nodes the demo builds, in order. */
export const RUNBOOK_STEPS: RunbookStepDef[] = [
  {
    type: "textInput",
    data: { label: "start", value: RUNBOOK_INPUT_TEXT, inputFields: [{ key: "text" }] },
    enterDelayMs: 250,
  },
  {
    type: "wait",
    data: { label: "wait", duration: 2000 },
    enterDelayMs: 700,
  },
  {
    type: "consoleLog",
    data: { label: "consoleLog", logMessage: RUNBOOK_LOG_EXPRESSION },
    enterDelayMs: 700,
  },
];

/** Pause (ms) after the graph is built before auto-running it. */
export const RUNBOOK_RUN_DELAY_MS = 600;

/** Build a positioned node for step `index`, marked for the entrance animation. */
export function buildRunbookNode(index: number): WorkflowNode {
  const step = RUNBOOK_STEPS[index];
  return {
    id: `node_runbook_${step.type}_${Date.now()}_${index}`,
    type: step.type,
    position: { x: index * NODE_DX, y: 0 },
    data: { ...step.data, __runbookEntrance: true },
  };
}

/** Build the edge connecting the previous node to this one. */
export function buildRunbookEdge(sourceId: string, targetId: string): WorkflowEdge {
  return {
    id: `edge_runbook_${sourceId}_${targetId}_${Date.now()}`,
    source: sourceId,
    target: targetId,
  };
}
```

## Task 4: Create `useRunbookPlayer.ts`

**Files:**
- Create: `frontend/src/features/runbook/useRunbookPlayer.ts`

Depends on: Task 3. Uses store members verified to exist: `addNode`, `addEdge`, `nodes`, `currentWorkflow`, `runInputValues`, `buildExecutionRequestBody`, `executeWorkflow`, `updateNode`.

- [ ] **Step 1: Write the composable**

```ts
import { ref } from "vue";
import { useRouter } from "vue-router";

import { useWorkflowStore } from "@/stores/workflow";
import { workflowApi } from "@/services/api";
import {
  RUNBOOK_INPUT_TEXT,
  RUNBOOK_QUERY_FLAG,
  RUNBOOK_QUERY_VALUE,
  RUNBOOK_RUN_DELAY_MS,
  RUNBOOK_STEPS,
  RUNBOOK_WORKFLOW_NAME,
  buildRunbookEdge,
  buildRunbookNode,
} from "@/features/runbook/runbookScript";
import type { WorkflowNode } from "@/types/workflow";

// Module-level singleton state: shared by every caller without a Pinia store.
const isRunbookPlaying = ref(false);

const prefersReducedMotion = (): boolean =>
  typeof window !== "undefined" &&
  window.matchMedia("(prefers-reduced-motion: reduce)").matches;

const sleep = (ms: number): Promise<void> =>
  new Promise((resolve) => window.setTimeout(resolve, ms));

export function useRunbookPlayer() {
  const router = useRouter();
  const workflowStore = useWorkflowStore();

  /** Create a fresh demo workflow and navigate to it with the auto-play flag. */
  async function startRunbookNewWorkflow(): Promise<void> {
    if (isRunbookPlaying.value) return;
    const workflow = await workflowApi.create({
      name: RUNBOOK_WORKFLOW_NAME,
      description: "Auto-generated Heym runbook demo.",
    });
    await router.push({
      name: "editor",
      params: { id: workflow.id },
      query: { [RUNBOOK_QUERY_FLAG]: RUNBOOK_QUERY_VALUE },
    });
    // EditorView consumes the query flag on mount and calls playRunbookInPlace().
  }

  /** Build the 3-node graph with entrance animation, then auto-run it. */
  async function playRunbookInPlace(): Promise<void> {
    if (isRunbookPlaying.value) return;
    if (!workflowStore.currentWorkflow) return;
    isRunbookPlaying.value = true;
    const reduced = prefersReducedMotion();

    try {
      const created: WorkflowNode[] = [];
      for (let i = 0; i < RUNBOOK_STEPS.length; i++) {
        if (!reduced) await sleep(RUNBOOK_STEPS[i].enterDelayMs);
        const node = buildRunbookNode(i);
        workflowStore.addNode(node);
        // addNode may rewrite data.label to keep it unique; re-read the stored node.
        const stored = workflowStore.nodes[workflowStore.nodes.length - 1];
        created.push(stored);
        if (i > 0) {
          workflowStore.addEdge(buildRunbookEdge(created[i - 1].id, stored.id));
        }
        // Clear the transient entrance flag after the slide-in finishes.
        if (!reduced) {
          window.setTimeout(() => {
            workflowStore.updateNode(stored.id, { __runbookEntrance: undefined });
          }, 700);
        } else {
          workflowStore.updateNode(stored.id, { __runbookEntrance: undefined });
        }
      }

      await sleep(reduced ? 0 : RUNBOOK_RUN_DELAY_MS);

      // Feed the sample input and run the real executor.
      workflowStore.runInputValues.text = RUNBOOK_INPUT_TEXT;
      await workflowStore.executeWorkflow(workflowStore.buildExecutionRequestBody());
    } finally {
      isRunbookPlaying.value = false;
    }
  }

  return { isRunbookPlaying, startRunbookNewWorkflow, playRunbookInPlace };
}
```
